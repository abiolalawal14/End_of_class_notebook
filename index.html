üíæ Module 5: MySQL Database Mastery

5.1 Powerful Query Techniques

JOINs - Connecting Your Data 
-- Find all orders placed by customers from Lagos
SELECT c.first_name, o.order_date, o.total_amount
FROM customers c
INNER JOIN orders o ON c.customer_id = o.customer_id
WHERE c.city = 'Lagos';
GROUP BY - Aggregating Data 
-- Calculate total sales for each city
SELECT city, SUM(o.total_amount) as total_sales
FROM customers c
LEFT JOIN orders o ON c.customer_id = o.customer_id
GROUP BY c.city
ORDER BY total_sales DESC;
